function calculate(){ let customer = document.getElementById("customer").value; let sft = Number(document.getElementById("sft").value); let rate = Number(document.getElementById("rate").value); let sellingPrice = Number(document.getElementById("gst").value); let miningQty = Number(document.getElementById("miningQty").value); let loading = Number(document.getElementById("loading").value); let rtgs = Number(document.getElementById("rtgs").value); let material = sft * rate; let gst = sellingPrice * 0.18; let mining = miningQty * 7; let total = material + gst + mining + loading; let axis = total - rtgs; document.getElementById("result").innerHTML = ` Customer : ${customer}

Material = ${material.toLocaleString()}
GST (18% of ${sellingPrice}) = ${gst.toLocaleString()}
Mining (${miningQty} × 7) = ${mining.toLocaleString()}
Loading = ${loading.toLocaleString()}

Total = ${total.toLocaleString()}
RTGS = ${rtgs.toLocaleString()}

Axis Amount = ${axis.toLocaleString()} `; }
